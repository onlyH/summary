  handleDeleteSelect() {
        let fromToIds = {}
        this.selectArray && this.selectArray.forEach(i => {
            let flag = true
            if (i.id1) {
                if (Object.keys(fromToIds).length > 0) {
                    for (let ids in fromToIds) {
                        if (ids == i.id1) {
                            fromToIds[ids].push(i.id2)
                            flag = false
                        }
                    }
                    if (flag) {
                        fromToIds[i.id1] = [i.id2]
                    }

                } else {
                    fromToIds[i.id1] = [i.id2]
                }

            } else {
                fromToIds[i.id] = i.objectInfo[0].id
            }
        })
        let params = {
            relType: this.$route.query.relationType,
            graphicType: this.$route.query.graphicType,
            fromToIds
        }
        relDel(params).then((res) => {
            if (res.result == "success") {
                this.$message.success("删除成功！");
                this.addRelationSucess(0)
            } else {
                this.$message.error("删除失败！");
            }
        });
        this.visible = false
    }
